65 Foam::tmp<Foam::volScalarField> Foam::GidaspowErgunWenYu::K
66 (
67  const volScalarField& Ur
68 ) const
69 {
70  volScalarField beta = max(scalar(1) - alpha_, scalar(1.0e-6));
71 
72  // volScalarField bp = pow(beta, -2.65);
73  volScalarField bp = pow(beta, -phasea_.hExp());
74  volScalarField Re = max(beta*Ur*phasea_.d()*phasea_.sF()/phaseb_.nu(),
75  scalar(1.0e-9));
76 
77  volScalarField Cds
78  (
79  neg(Re - 1000)*(24.0*(1.0 + 0.15*pow(Re, 0.687))/Re)
80  + pos(Re - 1000)*0.44
81  );
82 
83 /*
84  // Wen and Yu (1966)
85  // modified in May 17, 2012, by C.Z.
86  //tmp<volScalarField> tKWenYu = (0.75*Cds*phaseb_.rho()*Ur*bp
87  /(phasea_.d()*phasea_.sF()));
88  volScalarField tKWenYu = (0.75*Cds*phaseb_.rho()*Ur*bp
89  /(phasea_.d()*phasea_.sF()));
90  volScalarField& KWenYu = tKWenYu();
91 
92  // Ergun
93  forAll(beta, cellj)
94  {
95  if (beta[cellj] < 0.8)
96  {
97  tKWenYu[cellj] =
98  150.0*alpha_[cellj]*phaseb_.nu().value()*phaseb_.rho().value()
99  /sqr(beta[cellj]*phasea_.d().value())
100  + 1.75*phaseb_.rho().value()*Ur[cellj]
101  /(beta[cellj]*phasea_.d().value());
102  }
103  }
104 // WARNING: remove this line will makes the parallel computations "instable"
105  tKWenYu.correctBoundaryConditions();
106 
107  return KWenYu;
108 */
109 // pos0(beta - 0.8)*(0.75*Cds*phaseb_.rho()*Ur*bp
110 // /(phasea_.d()*phasea_.sF())) //line for sedfoam-5.0
111  return pos(beta - 0.8)
112  *(0.75*Cds*phaseb_.rho()*Ur*bp/(phasea_.d()*phasea_.sF()))
113  + neg(beta - 0.8)
114  *(150.0*alpha_*phaseb_.nu()*phaseb_.rho()/sqr(beta*phasea_.d())
115  + 1.75*phaseb_.rho()*Ur/(beta*phasea_.d()));
116 }
