56 Foam::tmp<Foam::volScalarField> Foam::granularRheologyModels::MuIdila::delta
57 (
58  const dimensionedScalar& K_dila,
59  const dimensionedScalar& alpha_c,
60  const volScalarField& alpha,
61  const volScalarField& magD,
62  const dimensionedScalar& da,
63  const dimensionedScalar& rhob,
64  const dimensionedScalar& nub,
65  const volScalarField& p_p_total,
66  const dimensionedScalar& PaMin
67 ) const
68 {
69  volScalarField I = magD*da/(pow(max(p_p_total, PaMin)/rhob, 0.5));
70  volScalarField alphaEq = (alpha_c)/(1.0+0.31*I);
71  return K_dila*(alpha-alphaEq);
72 }
